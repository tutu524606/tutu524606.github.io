    # 概念

    • 可重复执行特定任务的代码块或者事件触发时被调用执行的代码块
    // 相加功能
function add(a, b) {
  return a + b;
}
add(1, 2);
add(3, 4);
// 事件点击触发
document.body.onclick = function() {
  this.style.backgroundColor = "blue";
};

    # 创建

    • 函数对应的构造函数是Function，顾声明函数的关键字是function

    # 命名函数（function 声明函数）

    • 语法：function 函数名称(arg0,arg1,arg2,...){函数体}
    function fn() {
  console.log("函数体");
}

    # 匿名函数（函数表达式||构造函数实例）

    • 语法 1： new Function(参数字符串, 函数体字符串)
    var fn = new Function("", 'console.log("函数体");');
    • 语法 2： function(arg0,arg1,arg2,...){函数体}
    var fn = function() {
  console.log("函数体");
};

    # 参数

    • 形式参数形式参数出现在函数创建时
    function fn(a, b, c, d) {
  // 函数体
}
    • 实际参数实际参数出现在函数调用执行时
    function fn(a, b, c, d) {
  // 函数体
}

    • 同名参数取最后出现的那个
    function fn(a, a, a) {
  // 函数体
  console.log(a); // 3
}
fn(1, 2, 3);
    • 参数的省略形式参数和实际参数意义对应，个数不需要相同
    function fn(a, b, c) {
  //函数体
}
fn(1, 2);
fn(1, 2, 3, 4);
    • arguments 对象实际参数的集合，是一个类数组
    function fn() {
  console.log(arguments);
}
fn(1, 2, 3, 4);

    # 函数存储

    程序执行栈中同时存在同名的命名函数和匿名函数，会优先在执行位置向上查找，如果在执行栈中查找到栈内存中存在的匿名函数，则停止；否则，则去堆内存中查找是否存在 fn

    • 栈内存匿名函数存储在栈内存中

    • 堆内存命名函数存储在堆内存中

    console.log(fn); // function fn(){}
var fn = function() {};
function fn() {}

    # 函数调用

    程序在执行时，会产生一个执行栈，在执行栈中，程序的执行时自上而下的。在上述存储中，匿名函数存储在栈内存中，所以，匿名函数的执行存在执行顺序；而命名函数存储于堆内存中，对于堆内存的访问，则是通过地址指针去访问，所以，命名函数的执行与位置无关。

    • 命名函数调用位置解读
    fn(); // 函数执行
function fn() {
  console.log("函数体");
}
fn(); // 函数执行
    • 匿名函数调用位置解读
    fn(); // Uncaught TypeError: ff is not a function
console.log(ff); // 声明提升 undefined
var fn = function() {
  console.log("函数体");
};
ff(); // 函数执行

    # return 语句

    // 中断循环体
function fn() {
  return 1;
  console.log("该语句不执行");
}

// 函数返回值
function fn() {
  return 1;
}
fn(); // 函数的执行结果是1

// 函数体内终端整个嵌套循环
function fn() {
  for (var i = 0; i < 10; i++) {
    for (var j = 0; j < 10; j++) {
      console.log("函数体");
      return 1;
    }
  }
}
fn(); // 控制台只输出一次

// return 没有返回值时或者函数没有ruturn语句时，函数的执行结果是undefined
function fn() {}
fn(); // undefined

    # IIFE

    • JavaScript 的立即执行函数，简称 IIFE。可以使用立即执行函数模拟函数的块级作用域，在函数内部使用 var 声明的量，外部不可以访问（即 var 声明的量存在函数作用域），这样写的好处是，可以让量成为私有变量，避免全局的变量污染。
    (function() {
  var private = "私有变量";
})();
console.log(private); // Uncaught ReferenceError: private is not defined

    # 函数闭包

    • 概念：一个函数有权访问另外一个函数中的局部变量或者方法
    // 函数在执行完毕后，会销毁函数体内的局部变量和方法，而全局变量则会随着应用或者程序一直存在。
var counter = 1; // 全局变量
var fn = function() {
  return (counter += 1);
};

function add() {
  fn();
  return counter;
}
add(); // 2
add(); // 3

// 在上述例子中，如若我们想要函数的每次执行都让 i 加1，则需要使用到闭包
var add = (function() {
  var i = 1;
  return function() {
    return ++i;
  };
})();
add(); // 2
add(); // 3

    [^注解]：

    • 立即执行函数执行过后，设置i的值为1，并返回函数表达式赋值给add
    • 函数add在执行时，函数体内的变量i会从当前执行的函数体内向上查找
    • 最终在上一级的作用域中查找到，变量i是上一级作用域中的私有变量
    • 在IIFE立即执行函数关闭后，并不被销毁，原因是变量i在更下一层的作用域中被使用
    • 这种程序的执行方式，被称之为闭包。总结：只要存在函数作用域的嵌套，就是闭包

    # 高阶函数

    • 概念：通俗的将是函数作为参数传入到另外一个函数中调用执行。JavaScript 的函数其实是指向某个变量，既然变量可以指向函数，函数的参数能接收变量，那么一个函数就可以接收另一个函数作为参数，这种函数就称之为高阶函数。
    // 一个简单的高阶函数
function add(x, y, f) {
  return f(x) + f(y);
}
add(-5, 6, Math.abs);
    • 柯里化函数概念：将一个多参数的函数转换成多个单参数的函数，这个函数会返回一个函数去处理下一个参数。也就是把 fn(a,b,c)转换成 new Fn(a)(b)(c)这种形式。柯里化常见的应用有：参数复用，延迟计算。
    // 比如，拼接接口地址的函数：
function curry(fn) {
  let args = Array.prototype.slice.call(arguments, 1);
  return function() {
    let innerArgs = Array.prototype.slice.call(arguments);
    let finalArgs = args.concat(innerArgs);
    if (finalArgs.length < fn.length) {
      //fn.length为函数的参数个数
      return curry.call(this, fn, ...finalArgs);
    } else {
      return fn.apply(null, finalArgs);
    }
  };
}
var getAuthUrl = curry(getUrl, "http://localhost:8080/", "auth");
let loginUrl = getAuthUrl("/login");
let logoutUrl = getAuthUrl("/logout");

    // 柯里化函数之组合函数：组合函数类似于管道，多个函数执行时，上一个函数的返回值会自动传入到第二个函数继续执行。比如，替换一个url中的参数：
function replaceToken(str) {
  return str.replace(/{token}/g, "123455");
}
function replaceAccount(str) {
  return str.replace(/{account}/g, "xuriliang");
}
replaceAccount(
  replaceToken("http://localhost/api/login?token={token}&account={account}")
);

// 我们可以利用这种嵌套的写法来实现，但如果嵌套过多，代码可读性就不是很好了。当然我们也可以在一个函数里分过程实现，不过这样函数就不符合单一原则了。利用函数组合我们可以这样写：
function compose() {
  var args = arguments;
  var start = args.length - 1;
  return function() {
    var i = start;
    var result = args[start].apply(this, arguments);
    while (i--) result = args[i].call(this, result);
    return result;
  };
}
compose(
  replaceToken,
  replaceAccount
)("http://localhost/api/login?token={token}&account={account}");
