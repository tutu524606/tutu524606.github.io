# 数字对象 Number

# 全局方法

  • isNaN 判断是否是数字 NaN
// isNaN 返回结果是true和false，存在隐式类型转换
isNaN(1); // false
isNaN(NaN); // true
isNaN(null); // false
isNaN(false); // false
isNaN(undefined); // true
isNaN(""); // false
  • parseInt 转换成整型数字
// parseInt 返回整型数字，非数字返回NaN
parseInt(1); // 1
parseInt(NaN); // NaN
parseInt(null); // NaN
parseInt(false); // NaN
parseInt(undefined); // NaN
parseInt(3.6); // 3
parseInt("100.001"); // 100
parseInt("100px"); // 100
  • parseFloat 转换成浮点型数字
// parseFloat 返回整型数字，非数字返回NaN
parseFloat(1); // 1
parseFloat(NaN); // NaN
parseFloat(null); // NaN
parseFloat(false); // NaN
parseFloat(undefined); // NaN
parseFloat(3.6); // 3.6
parseFloat("100.001"); // 100.001
parseFloat("100.1px"); // 100.1
  • Number 转换成数字数字的强制转换函数，在数据类型转换章节已讲解

# 静态方法

  • Number.isFinite() 判断是否是有理数
  • Number.isNaN() 判断是否是 NaN
  • Number.isInteger() 判断是否是整型数字
  • Number.parseInt() 转换整型数字
  • Number.parseFloat() 转换浮点型数字

# 原型方法 Number.prototype

  • *.toFixed(n) 保留小数位，遵循四舍五入
  • *.toPrecision() 保留有效位数字，遵循四舍五入原则
var num = 3.1415926;
num.toFixed(3); // 3.142
num.toPrecision(4); // 3.142

# 数据交换对象 JSON

  • 定义一个对象，下面序列化和反序列化的依据对象
var people = {
  name: "小明",
  age: 18,
  sex: true,
  height: 180,
  grade: null,
  "high-school": "加里敦大学-Jiali Dun",
  skills: ["JavaScript", "Vue", "Nodejs", "PHP"],
};
  • 序列化
var s = JSON.stringify(people);
console.log(s); // 序列化成JSON格式的字符串
/*
{"name":"小明","age":18,"sex":true,"height":180,"grade":null,"high-school":"加里敦大学-Jiali Dun","skills":["JavaScript","Vue","Nodejs","PHP"]}
*/

// 想要输出的美观一些，可以加上参数，按缩进输出
var format_s = JSON.stringify(people, null, "  ");
console.log(format_s);
/*
{ "name": "小明", "age": 18, "sex": true, "height": 180, "grade": null, "high-school": "加里敦大学-Jiali Dun", "skills": [ "JavaScript", "Vue", "Nodejs", "PHP" ] }
*/

// 序列化方法的第二个参数用于控制如何筛选对象的键值，若想要输出指定的属性，可以传入数组
var filter_s = JSON.stringify(people, ["name", "skills"], "  ");
console.log(filter_s);
/*
{ "name": "小明", "skills": [ "JavaScript", "Vue", "Nodejs", "PHP" ] }
*/

// 还可以传入一个函数，对象的每个键值会被函数优先处理
function covert(key, value) {
  if (typeof value === "string") {
    return value.toUpperCase();
  }
  return value;
}
var covert_s = JSON.stringify(people, covert, "  ");
console.log(covert_s);
/*
{ "name": "小明", "age": 18, "sex": true, "height": 180, "grade": null, "high-school": "加里敦大学-JIALI DUN", "skills": [ "JAVASCRIPT", "VUE", "NODEJS", "PHP" ] }
*/

// 精确的控制序列化后的数据，可以使用toJSON方法
var peopleToJSON = {
  name: "小明",
  age: 18,
  sex: true,
  height: 180,
  grade: null,
  "high-school": "加里敦大学-Jiali Dun",
  skills: ["JavaScript", "Vue", "Nodejs", "PHP"],
  toJSON: function() {
    return {
      Name: this.name,
      Age: this.age,
    };
  },
};
var toJSON_s = JSON.stringify(peopleToJSON);
console.log(toJSON_s);
/*
{"Name":"小明","Age":18}
*/
  • 反序列化
// 使用JSON.parse把JSON格式的字符串转换成JavaScript对象
JSON.parse("[1,2,3,true]"); // [1,2,3,true]
JSON.parse('{"name": "小明", "age": "18"}'); // {age: '小明', age: 18}
JSON.parse("false"); // false

// 反序列化时，也可以接收一个函数，用来转换解析出的属性
var p = JSON.parse(JSON.stringify(people), function(key, value) {
  if (key == "name") {
    return value + "同学";
  }
  return value;
});
console.log(JSON.stringify(p));
/*
{"name":"小明同学","age":18,"sex":true,"height":180,"grade":null,"high-school":"加里敦大学-Jiali Dun","skills":["JavaScript","Vue","Nodejs","PHP"]}
*/
