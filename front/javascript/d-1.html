    # 数据类型

    类型 值 typeof 检测结果 对应构造函数
    字符串 单引号或者双引号引起来的内容 'string' String
    数字 NaN、整数、浮点数、Infinity、-Infinity 'number' Number
    布尔 true、false 'boolean' Boolean
    空对象 null 'object' 无
    未定义 undefined 'undefined' 无
    对象 {} 'object' Object

    # 创建

    • 字符串的创建
    var str = "abc"; // 字面量方式创建字符串
var str1 = new String("abc").valueOf(); // new 关键字实例化构造函数的方式创建字符串
    • 数字的创建
    var num = 10; // 字面量方式创建数字
var num1 = new Number(10).valueOf(); // new 关键字实例化构造函数的方式创建数字
    • 布尔的创建
    var num = true; // 字面量方式创建布尔
var num1 = new Boolean(true).valueOf(); // new 关键字实例化构造函数的方式创建布尔
    • 对象的创建
    var obj = {}; // 字面量方式创建对象
var obj1 = new Object(); // new 关键字实例化构造函数的方式创建对象

    # 数据类型转换

    # 1. 转字符串

    • 使用 toString 方法转换
    var num = 10;
console.log(num.toString()); // '10'
var bool = true;
console.log(bool.toString()); // 'true'
var n = null,
  u = undefined;
console.log(n.toString()); // Uncaught TypeError: Cannot read property 'toString' of null
console.log(u.toString()); // Uncaught TypeError: Cannot read property 'toString' of null
    • 使用 String 强转函数
    var num = 10,
  bool = true,
  n = null,
  u = undefined;
console.log(String(num), String(bool), String(n), String(u)); // '10' 'true' 'null' 'undefined'

    # 2. 转数字

    • 使用 parseInt 转整形数字
    var num1 = 10,
  num2 = "20",
  num3 = 3.1415926,
  num4 = "6.68686868",
  unit = "100px",
  str = "abc",
  bool = true,
  n = null,
  u = undefined;
console.log(
  parseInt(num1),
  parseInt(num2),
  parseInt(num3),
  parseInt(num4),
  parseInt(unit),
  parseInt(str),
  parseInt(bool),
  parseInt(n),
  parseInt(u)
);
/*10 20 3 6 100 NaN NaN NaN NaN*/
    • 使用 parseFloat 转浮点型数字
    var num1 = 10,
  num2 = "20",
  num3 = 3.1415926,
  num4 = "6.68686868",
  unit = "100px",
  str = "abc",
  bool = true,
  n = null,
  u = undefined;
console.log(
  parseFloat(num1),
  parseFloat(num2),
  parseFloat(num3),
  parseFloat(num4),
  parseFloat(unit),
  parseFloat(str),
  parseFloat(bool),
  parseFloat(n),
  parseFloat(u)
);
/*10 20 3.1415926 6.68686868 100 NaN NaN NaN NaN*/
    • 使用 Number 强转函数
    var num = "100",
  str = "",
  str1 = "abc",
  unit = "100px",
  bool = true,
  n = null,
  u = undefined;
console.log(
  Number(num),
  Number(str),
  Number(str1),
  Number(unit),
  Number(bool),
  Number(n),
  Number(u)

/*100 0 NaN NaN 1 0 NaN*/

    # 3. 转布尔

    • 布尔类型只有两个值，true 和 false。在 JavaScript 中，只有 7 个假值，除此之外的所有值都为真值
    • 7 个假值：false、NaN、null、undefined、''、0、-0
    • 所有的对象都是真值
    • 所有的函数或者方法都是真值
    • 使用逻辑非操作符转换
    console.log(!!{}, !!String, !!0, !!"a", !!""); // true  true  false true  false
    • 使用 Boolean 强转函数
    console.log(
  Boolean({}),
  Boolean(String),
  Boolean(0),
  Boolean("a"),
  Boolean("")
); // true  true  false true  false

    # 4. JSON 对象转换

    • 普通对象转 json 对象
    var obj = {
  name: "张三",
  age: 18,
  sex: true,
  wife: null,
  son: undefined,
  identity: {
    card: "12321321",
    address: "地球",
  },
};
console.log(JSON.stringify(obj)); // {"name":"张三","age":18,"sex":true,"wife":null,"identity":{"card":"12321321","address":"地球"}}
    • json 对象转普通对象
    var json =
  '{"name":"张三","age":18,"sex":true,"wife":null,"identity":{"card":"12321321","address":"地球"}}';
console.log(JSON.parse(json));
/*
{
  name: "张三",
  age: 18,
  sex: true,
  wife: null,
  son: undefined,
  identity: {
    card: "12321321",
    address: "地球",
  },
}
*/

    # 5. 隐式类型转换

    // 隐式类型转换有很多，下面简单列举几个
var str = 10 + "10";
console.log(str); // '1010'

var num = "" * "10";
console.log(num); // 0

console.log(!"", !10); // true false

console.log("10" > "9"); // false

console.log(null >= -1); // true
