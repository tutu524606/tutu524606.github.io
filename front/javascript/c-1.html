# 流程控制语句

# if 语句

# 单分支语句

  • if(条件表达式){代码块}
var laber = 100;
if (laber < 50) {
  alert("laber大于50");
  alert("不管怎样，我都能被执行到！"); //用复合语句包含，判断后执行一条复合语句
}

# 双分支语句

  • if(条件表达式){代码块}else{代码块}
var laber = 100;
if (laber > 50) {
  alert("laber大于50"); //条件为true，执行这个代码块
} else {
  alert("laber小于50"); //条件为false，执行这个代码块
}

# 多分支语句

  • if(条件表达式){代码块}else if(条件表达式){代码块}...else{代码块}
var laber = 100;
if (laber >= 100) {
  //如果满足条件，不会执行下面任何分支
  alert("甲");
} else if (laber >= 90) {
  alert("乙");
} else if (laber >= 80) {
  alert("丙");
} else if (laber >= 70) {
  alert("丁");
} else if (laber >= 60) {
  alert("及格");
} else {
  //如果以上都不满足，则输出不及格
  alert("不及格");
}

# switch 语句

1. switch 语句是多重条件判断，用于多个值相等的比较
2. 分支结构过多时，不建议使用 if 多分支
3. switch 语句的条件表达式的运算结果和 case 后面的条件表达式的运算结果进行全等比较

var laber = 1;
switch (
  laber //用于判断box相等的多个值
) {
  case 1:
    alert("one");
    break; //break;用于防止语句的穿透
  case 2:
    alert("two");
    break;
  case 3:
    alert("three");
    break;
  default:
    //相当于if语句里的else，否则的意思
    alert("error");
}

[^注解]：语法结构中的 break 切记不可漏写，除非是程序执行环境需要。break 不写造成的后果将是条件成立后，后面 case 中的代码块也会执行。

# 循环语句

# for 语句

概念：是一种先判断，后运行的循环语句
语法：for(初始值;判断条件;步长更新){代码块}
场景：循环次数固定时使用

for (var laber = 1; laber <= 5; laber++) {
  //第一步，声明变量var laber = 1;
  alert(laber); //第二步，判断laber <=5
}

var laber = 1;
for (; laber <= 5; laber++) {
  //第一步，声明变量var laber = 1;
  alert(laber); //第二步，判断laber <=5
}

/*多个判断条件，以最后一个判断条件为标准*/
var i = 1,
  j = 10;
for (; i <= 5, i >= 2; i++, j--) {
  console.log(i, j);
}

/*死循环*/
for (;;) {}
for (var i = 0; i >= 0; i++) {}
for (var i = 10; i <= 10; i--) {}

# while 语句

概念：是一种先判断，后运行的循环语句
语法：while(判断条件){代码块}
场景：循环次数不固定时使用

var laber = 1; //如果是1，执行五次，如果是10，不执行
while (laber <= 5) {
  //先判断，再执行
  alert(laber);
  laber++;
}

# do...while 语句

概念：是一种先运行，后判断的循环语句
语法：do(判断条件)while{代码块}
场景：循环次数不固定时使用

var laber = 1; //如果是1，执行五次，如果是10，执行1次
do {
  alert(laber);
  laber++;
} while (laber <= 5);

# for...in 语句

概念：是一种精准的迭代语句，遍历对象的可枚举属性
语法：for(属性 in 对象){代码块}
场景：遍历对象时使用

var laber = {
  //创建一个对象
  name: "moxiaobo", //键值对，左边是属性名，右边是值
  age: 28,
  height: 178,
};
for (var p in laber) {
  //列举出对象的所有属性
  alert(p);
}

# 结构控制语句

# break 和 continue 语句

概念：break 语句断开当前循环

for (var laber = 1; laber <= 10; laber++) {
  if (laber == 5) break; //如果laber 是5，就退出循环
  document.write(laber);
  document.write("<br />");
}

概念：continue 语句调出单前循环，继续下一次循环

for (var laber = 1; laber <= 10; laber++) {
  if (laber == 5) continue; //如果laber 是5，就退出当前循环
  document.write(laber);
  document.write("<br />");
}

需求：通过改变最外层循环条件，断开整个嵌套循环

for (var i = 0; i < 10; i++) {
  for (var j = 0; j < 10; j++) {
    i = 11;
  }
}

# throw 异常处理语句

概念：throw 允许创建自定义错误

var num = "10";
if (typeof num != "number") throw "num必须为数字";

# 异常捕获与处理

概念：用于处理代码中可能出现的错误信息。当存在错误时，对错误进行处理，并且不影响程序的执行。 语法：try{}catch(e){}finally{}

  • try 语句允许我们定义在执行时进行错误测试的代码块。

  • catch 语句允许我们定义当 try 代码块发生错误时，所执行的代码块。

  • finally 语句在 try 和 catch 之后无论有无异常都会执行。

try {
  alerttt("错误执行语句");
} catch (e) {
  throw new Error(e.message);
} finally {
  console.log("最终执行语句");
}
console.log("出现错误时，语句继续执行");

[^注解]：catch 和 finally 语句都是可选的，但你在使用 try 语句时必须至少使用一个。
[^提示]： 当错误发生时， JavaScript 会停止执行，并生成一个错误信息。使用 throw 语句 来创建自定义消息(抛出异常)。如果你将 throw 和 try 、 catch 一起使用，就可以控制程序输出的错误信息。
